set1 = {2,2.2,True,4+3j}
set2 = {1,2.2,False,4+3j}
print(set1&set2)  # 交集，因为int(True) == 1，在set2中找共同的
# {1, 2.2, (4+3j)}
print(set2&set1)  # 交集，因为bool(1) = True，在set1中找共同的
# {True, 2.2, (4+3j)}
print(set1|set2)  # 并集，因为int(True) == 1，在set1中找相同的
# {False, True, 2, 2.2, (4+3j)}
print(set2|set1)  # 并集，因为bool(1) = True，在set2中找相同的
# {False, 1, 2.2, 2, (4+3j)}
print(set1-set2)  # 差集，因为int(True) == 1，在set2中找set1里没有的
# {2}
print(set2-set1)  # 差集，因为bool(1) = True，在set1中找set2里没有的
# {False}

    j = 1

n = 0
for n1 in range(1, 6):
    for n2 in range(1, 6):
        for n3 in range(1, 6):
            if n1 != n2 and n2 != n3 and n1 != n3:  
                n += 1
print(n)
# 递归计数法（优点：参数自由控制，可以解决同一类的任意参数；缺点：耗脑子）
def loop(nums, deep):  
    global n
    if not deep:
        n += 1
    for num in nums:
        loop(nums - {num}, deep - 1)
n = 0
loop(set(range(1, 6)), 3)
print(n)
# 集合去重法
n = 0
for i in range(100, 600):
    if not set(str(i)) - {'1', '2', '3', '4', '5'} and len(set(str(i))) == 3:
        n += 1

二进制妙用

例如：
输入：1，2，3

# 递归法
def loop(nums):
    if nums:
        yield ''.join(sorted(nums))
        for num in nums:
            yield from loop(nums - {num})
s = '1,2,3'
print(sorted(set(loop(set(s.split(','))))))
# 二进制妙用
s = '1,2,3'
lst = s.split(',')
big_result = []
for i in range(1, 2**len(lst)):
    ss = format(i, f'0{len(lst)}b')
    result = []
    for j in range(len(ss)):
        if ss[j] == '1':
            result.append(lst[j])
    big_result.append(int(''.join(result)))
print(sorted(big_result))

62进制

写一个base62encode函数，计算62进制
即：0123456789ABCD...Zabcd...z（10个数字，26个大写字母，26个小写字母）
例子：
base62encode(1) = 1
base62encode(61) = z
base62encode(62) = 10

def base62encode(n):
    lst = []
    s = '0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz'
    while n > 0:
        y = s[n % 62]
        lst.append(y)
        n = n // 62
    return ''.join(lst[::-1])
print(base62encode(123))

import time
def wrapper(function):
    start = 0
    def inner(*args, **kwargs):
        nonlocal start
        now = time.time()
        if now - start > 10:
            ret = function(*args, **kwargs)
            start = now
            return ret
        else:
            print(f'对不起，您访问的过于频繁，请于{10 - int(now - start)}秒之后访问')
    return inner

如何精确计算开平方小数位数

# 手动开平方法（point控制小数位数）
def pow_big(num, point: int = 0):  
    num = str(num)[::-1]
    nums = []
    res = []
    for index in range(len(num[::2])):
        nums.append(num[index * 2: index * 2 + 2][::-1])
    nums = nums[::-1]
    for index in range(len(nums)):
        temp1 = int(''.join(res) or 0) * 20
        temp2 = int(''.join(nums[index - 1 if index else 0: index + 1]))
        for n in range(11):
            if (temp1 + n) * n > temp2:
                res.append(str(n - 1))
                nums[index] = str(temp2 - (temp1 + n - 1) * (n - 1))
                break
            elif (temp1 + n) * n == temp2:
                res.append(str(n))
                nums[index] = str(temp2 - (temp1 + n) * n)
                break
    if point > 0:
        res2 = []
        decimal = int(nums[-1])
        while point:
            temp1 = float(''.join(res + res2) or 0) * 20
            temp2 = decimal * 100
            for n in range(11):
                if (temp1 + n) * n > temp2:
                    res2.append(str(n - 1))
                    decimal = temp2 - (temp1 + n - 1) * (n - 1)
                    break
                elif (temp1 + n) * n == temp2:
                    res2.append(str(n))
                    decimal = temp2 - (temp1 + n) * n
                    break
            point -= 1
        return ''.join(res) + '.' + ''.join(res2)
    else:
        return int(''.join(res))
print(pow_big(x, 25))
